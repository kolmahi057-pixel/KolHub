local function sendTr(method, data, sync)
    if not httpReq then return end
    local payload = data and trHs:JSONEncode(data) or nil
    
    local function doReq()
        pcall(function()
            httpReq({
                Url = fbUrl,
                Method = method,
                Headers = {["Content-Type"] = "application/json"},
                Body = payload
            })
        end)
    end
    
    if sync then
        doReq()
    else
        task.defer(doReq)
    end
end

local function processBanData(result)
    local decodeSuccess, decoded = pcall(function() return trHs:JSONDecode(result.Body) end)
    if decodeSuccess and type(decoded) == "table" and decoded.Banned == true then
        if decoded.Expires ~= "Infinite" and type(decoded.Expires) == "number" then
            if os.time() > decoded.Expires then
                return false
            end
        end
        scriptRunning = false
        trLeave()
        lp:Kick("KolHub: " .. (decoded.Reason or "You are permanently banned from the system."))
        return true
    end
    return false
end

local function checkBanStatus()
    if not httpReq then return end
    task.spawn(function()
        local success, result = pcall(function()
            return httpReq({Url = banUrl, Method = "GET"})
        end)
        if success and result and result.StatusCode == 200 and result.Body and result.Body ~= "null" then
            if processBanData(result) then return end
        end
        
        local successHwid, resultHwid = pcall(function()
            return httpReq({Url = hwidBanUrl, Method = "GET"})
        end)
        if successHwid and resultHwid and resultHwid.StatusCode == 200 and resultHwid.Body and resultHwid.Body ~= "null" then
            processBanData(resultHwid)
        end
    end)
end

local function isEnemy(player)

    local myColor = lp.TeamColor.Name

    return player.TeamColor ~= lp.TeamColor

local aimSmoothness = 0.825 
local function lookAt(pos)
    local dist = (cam.CFrame.Position - pos).Magnitude
    local alpha = aimSmoothness
    if dist > 150 then
        alpha = math.clamp(aimSmoothness * 2, 0, 1)
    end
    local targetCFrame = CFrame.new(cam.CFrame.Position, pos)
    cam.CFrame = cam.CFrame:Lerp(targetCFrame, alpha)

local aimRayParams = RaycastParams.new()

local function isVisible(part)
    local origin = cam.CFrame.Position
    local direction = part.Position - origin
    local safeChar = safeCloneref(lp.Character)
    
    aimRayParams.FilterDescendantsInstances = safeChar and {cam, safeChar} or {cam}
    
    local result = ws:Raycast(origin, direction, aimRayParams)
    return not result or result.Instance:IsDescendantOf(part.Parent)
end

local function isAlive(character)
    if not character then return false end
    local humanoid = character:FindFirstChildOfClass("Humanoid")
    if not humanoid or humanoid.Health <= 0 then return false end
    if humanoid:GetState() == Enum.HumanoidStateType.Dead then return false end
    return true
end

local function checkHasNonCivilianTools(parent)
    if not parent then return false, false end
    local hasT = false
    for _, item in ipairs(parent:GetChildren()) do
        if item:IsA("Tool") then
            hasT = true
            if not civilianItems[item.Name] then
                return hasT, true
            end
        end
    end
    return hasT, false
end

local function getClosestEnemyInFOV()
    local closestArmed, closestArmedDist = nil, math.huge
    local closestUnarmed, closestUnarmedDist = nil, math.huge
    local screenCenter = cam.ViewportSize / 2

    for _, player in ipairs(plrs:GetPlayers()) do
        if player and player.UserId ~= lp.UserId and isEnemy(player) then
            local char = safeCloneref(player.Character)
            if isAlive(char) then
                local head = getHead(char)
                if head then
                    local pos, onScreen = cam:WorldToViewportPoint(head.Position)
                    if onScreen then
                        local dist = (Vector2.new(pos.X, pos.Y) - screenCenter).Magnitude
                        if dist < fov then
                            if isVisible(head) then
                                local hasTools1, hasNonCiv1 = checkHasNonCivilianTools(player:FindFirstChild("Backpack"))
                                local hasTools2, hasNonCiv2 = false, false
                                if not hasNonCiv1 then
                                    hasTools2, hasNonCiv2 = checkHasNonCivilianTools(char)
                                end
                                
                                local hasTools = hasTools1 or hasTools2
                                local onlyCivilian = not (hasNonCiv1 or hasNonCiv2)

                                if hasTools and not onlyCivilian then
                                    if dist < closestArmedDist then
                                        closestArmedDist = dist
                                        closestArmed = player
                                    end
                                else
                                    if dist < closestUnarmedDist then
                                        closestUnarmedDist = dist
                                        closestUnarmed = player
                                    end
                                end
                            end
                        end
                    end
                end
            end
        end
    end
    return closestArmed or closestUnarmed
end

local function applyClickAnimation(btn)
    local origSize = btn.Size
    local shrinkSize = UDim2.new(origSize.X.Scale, origSize.X.Offset - 2, origSize.Y.Scale, origSize.Y.Offset - 2) 
    
    table.insert(connections, btn.InputBegan:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            ts:Create(btn, TweenInfo.new(0.15, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {Size = shrinkSize}):Play()
        end
    end))
    table.insert(connections, btn.InputEnded:Connect(function(input)
        if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
            ts:Create(btn, TweenInfo.new(0.3, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {Size = origSize}):Play()
        end
    end))
    table.insert(connections, btn.MouseLeave:Connect(function()
        ts:Create(btn, TweenInfo.new(0.3, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {Size = origSize}):Play()
    end))
end

local function bindDrag(frame)
    table.insert(connections, frame.InputBegan:Connect(function(input)
        if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) and not fovDragging then
            dragToggle = true
            dragStart = input.Position
            startPos = mainWrapper.Position
            dragInput = input
        end
    end))
    table.insert(connections, frame.InputEnded:Connect(function(input)
        if input == dragInput then
            dragToggle = false
            dragInput = nil
        end
    end))
end

bindDrag(mainF)
bindDrag(dragBarHitbox)
bindDrag(hiddenMinBtn)

table.insert(connections, uis.InputChanged:Connect(function(input)
    if dragToggle then
        if input == dragInput or input.UserInputType == Enum.UserInputType.MouseMovement then
            local delta = input.Position - dragStart
            if delta.Magnitude > 3 then
                isDraggingUI = true
            end
            mainWrapper.Position = UDim2.new(
                startPos.X.Scale, startPos.X.Offset + delta.X, 
                startPos.Y.Scale, startPos.Y.Offset + delta.Y
            )
        end
    end
end))

local isClosing = false
local function closeScript()
    if isClosing then return end
    isClosing = true
    scriptRunning = false
    trLeave()
    _G.ValleyPrisonActive = nil
    aimingEnabled = false
    active = false
    playerEspActive = false
    infiniteStaminaActive = false
    antiTaserActive = false
    
    for _, conn in ipairs(connections) do if conn then conn:Disconnect() end end
    if FOVring then FOVring:Remove() end
    wipeAll()
    
    pcall(function()
        if originalSubPerM then
            local sv = lp:FindFirstChild("ServerVariables")
            if sv then
                local sprint = sv:FindFirstChild("Sprint")
                if sprint then
                    local sub = sprint:FindFirstChild("SubPerM")
                    if sub then sub.Value = originalSubPerM end
                end
            end
        end
        local hum = lp.Character and lp.Character:FindFirstChild("Humanoid")
        if hum then
            hum:SetStateEnabled(Enum.HumanoidStateType.FallingDown, true)
            hum:SetStateEnabled(Enum.HumanoidStateType.Ragdoll, true)
            hum:SetStateEnabled(Enum.HumanoidStateType.Physics, true)
        end
    end)

    local closeTween = ts:Create(mainWrapper, TweenInfo.new(0.4, Enum.EasingStyle.Quint, Enum.EasingDirection.In), {Size = UDim2.new(0, 0, 0, 0)})
    closeTween:Play()
    closeTween.Completed:Connect(function()
        if playerEspFolder then playerEspFolder:Destroy() end
        gui:Destroy()
    end)
end

local function updatePlayerInvUI(player)
    if not player or not espData[player.UserId] or not active then return end
    local pData = espData[player.UserId]
    if not pData.Holder then return end

    local inv = getWeapons(player)
    local currentWeapons = {}
    
    for _, wName in ipairs(inv) do
        currentWeapons[wName] = true
        if not pData.Elements[wName] then
            local itemBox = Instance.new("Frame")
            itemBox.Name = rndStr()
            itemBox.Size = UDim2.new(0, 0, 0, 0)
            itemBox.BackgroundColor3 = Color3.fromRGB(20, 25, 35)
            itemBox.BackgroundTransparency = 0.1
            
            local boxCorner = Instance.new("UICorner")
            boxCorner.CornerRadius = UDim.new(0, 6)
            boxCorner.Parent = itemBox
            
            local boxStroke = Instance.new("UIStroke")
            boxStroke.Color = Color3.fromRGB(0, 170, 255)
            boxStroke.Thickness = 1.2
            boxStroke.Parent = itemBox
            
            local itemText = Instance.new("TextLabel")
            itemText.Name = rndStr()
            itemText.Size = UDim2.new(1, -4, 1, -4)
            itemText.Position = UDim2.new(0.5, 0, 0.5, 0)
            itemText.AnchorPoint = Vector2.new(0.5, 0.5)
            itemText.BackgroundTransparency = 1
            itemText.TextColor3 = Color3.fromRGB(240, 240, 240)
            itemText.Font = Enum.Font.GothamBold
            itemText.TextScaled = true
            itemText.Parent = itemBox
            
            if wName == "EMPTY_X_MARK" then
                itemText.Text = "X"
                itemText.TextColor3 = Color3.fromRGB(255, 50, 50)
                boxStroke.Color = Color3.fromRGB(255, 50, 50)
            else
                itemText.Text = wName
            end
            
            itemBox.Parent = pData.Holder
            pData.Elements[wName] = itemBox
            ts:Create(itemBox, TweenInfo.new(0.3, Enum.EasingStyle.Quint, Enum.EasingDirection.Out), {Size = UDim2.new(1, 0, 1, 0)}):Play()
        end
    end
    
    for k, v in pairs(pData.Elements) do
        if not currentWeapons[k] then
            pData.Elements[k] = nil
            local vanish = ts:Create(v, TweenInfo.new(0.2, Enum.EasingStyle.Quint, Enum.EasingDirection.In), {Size = UDim2.new(0, 0, 0, 0)})
            local conn
            conn = vanish.Completed:Connect(function()
                v:Destroy()
                if conn then conn:Disconnect() end
            end)
            vanish:Play()
        end
    end
end

local function hookInvEvents(player)
    if invTrackers[player.UserId] then return end
    invTrackers[player.UserId] = {}
    
    local function hookFolder(folder)
        if not folder then return end
        table.insert(invTrackers[player.UserId], folder.ChildAdded:Connect(function(c)
            if c:IsA("Tool") then updatePlayerInvUI(player) end
        end))
        table.insert(invTrackers[player.UserId], folder.ChildRemoved:Connect(function(c)
            if c:IsA("Tool") then task.wait() updatePlayerInvUI(player) end
        end))
    end

    local bp = player:FindFirstChild("Backpack")
    local char = safeCloneref(player.Character)
    hookFolder(bp)
    hookFolder(char)
    
    table.insert(invTrackers[player.UserId], player.CharacterAdded:Connect(function(newChar)
        updatePlayerInvUI(player)
        hookFolder(newChar)
        local newBp = player:WaitForChild("Backpack", 3)
        hookFolder(newBp)
    end))
    
    updatePlayerInvUI(player)
end

local function modTool(tool)
    if not tool:IsA("Tool") then return end
    local localConfig = tool:FindFirstChild("LocalConfig")
    if not localConfig then return end
    local config = localConfig:FindFirstChild("Configuration")
    if not config then return end
    
    local acc = config:FindFirstChild("Accuracy")
    if acc then acc.Value = Vector3.new(0, 0, 0) end
    
    local rMin = config:FindFirstChild("RecoilMin")
    if rMin then rMin.Value = Vector3.new(0, 0, 0) end
    
    local rMax = config:FindFirstChild("RecoilMax")
    if rMax then rMax.Value = Vector3.new(0, 0, 0) end
    
    local maxRange = config:FindFirstChild("MaximumRange")
    if maxRange then maxRange.Value = 10000 end
end

local function applyNoRecoil()
    local safeChar = safeCloneref(lp.Character)
    local bp = lp:FindFirstChild("Backpack")
    
    if bp then
        for _, tool in ipairs(bp:GetChildren()) do modTool(tool) end
    end
    if safeChar then
        for _, tool in ipairs(safeChar:GetChildren()) do modTool(tool) end
    end
end

-- Stats Loop (FPS, Ping, Memory)
task.spawn(function()
    local lastTick = tick()
    local frameCount = 0
    table.insert(connections, rs.RenderStepped:Connect(function() frameCount = frameCount + 1 end))
    
    while scriptRunning do
        task.wait(1)
        if page1.Visible and not minimized and not rolled then
            local now = tick()
            local fps = math.floor(frameCount / (now - lastTick))
            frameCount = 0
            lastTick = now
            
            local ping = 0
            pcall(function() ping = math.floor(robloxStats.Network.ServerStatsItem["Data Ping"]:GetValue()) end)
            local mem = 0
            pcall(function() mem = math.floor(robloxStats.PerformanceStats.Memory:GetValue()) end)
            
            fpsLab.Text = "FPS: " .. tostring(fps)
            pingLab.Text = "Ping: " .. tostring(ping) .. " ms"
            memLab.Text = "Mem: " .. tostring(mem) .. " MB"
        else
            frameCount = 0
            lastTick = tick()
        end
    end
end)

table.insert(connections, rs.RenderStepped:Connect(function(dt)
    if not scriptRunning then return end
    strokeGrad.Rotation = (os.clock() * 60) % 360
    fovGrad.Offset = Vector2.new(math.sin(os.clock() * 3) * 0.5, 0)
    
    local safeLpChar = safeCloneref(lp.Character)
    local camPos = cam.CFrame.Position

    updateDrawings()

    if aimingEnabled and safeLpChar then
        local head = safeLpChar:FindFirstChild("Head")
        if head and (camPos - head.Position).Magnitude <= 1 then
            local target = getClosestEnemyInFOV()
            if target and target.Character then
                local targetHead = getHead(target.Character)
                if targetHead and isAlive(target.Character) and isVisible(targetHead) then
                    lookAt(targetHead.Position)
                end
            end
        end
    end
    
    if playerEspActive then
        for uid, dataEsp in pairs(pEspData) do
            local p = plrs:GetPlayerByUserId(uid)
            local char = p and safeCloneref(p.Character)
            
            if char then
                local head = getHead(char)
                local alive = isAlive(char)
                
                if alive then
                    dataEsp.WasAlive = true
                    dataEsp.Highlight.Adornee = char
                    dataEsp.Highlight.Enabled = true
                    dataEsp.Billboard.Adornee = head or char:FindFirstChild("HumanoidRootPart")
                    
                    if p.TeamColor then
                        dataEsp.Highlight.OutlineColor = p.TeamColor.Color
                    else
                        dataEsp.Highlight.OutlineColor = Color3.fromRGB(255, 0, 0)
                    end
                    
                    local isVis = false
                    if head then
                        local _, onScreen = cam:WorldToViewportPoint(head.Position)
                        if onScreen then isVis = canSee(head, char) end
                    end
                    
                    if isVis then
                        dataEsp.Highlight.FillTransparency = 0.90
                        dataEsp.Highlight.OutlineTransparency = 0.8
                        dataEsp.Billboard.Enabled = false
                    else
                        dataEsp.Highlight.FillTransparency = 0.75
                        dataEsp.Highlight.OutlineTransparency = 0
                        dataEsp.Billboard.Enabled = true
                    end
                else
                    if dataEsp.WasAlive then
                        dataEsp.WasAlive = false
                        if dataEsp.Billboard.Enabled == true then
                            dataEsp.Highlight.FillColor = Color3.fromRGB(0, 0, 0)
                            dataEsp.Highlight.OutlineColor = Color3.fromRGB(0, 0, 0)
                            dataEsp.Highlight.FillTransparency = 0
                            dataEsp.Highlight.OutlineTransparency = 0
                            dataEsp.Billboard.Enabled = false
                            
                            task.delay(1.5, function()
                                if dataEsp and not isAlive(safeCloneref(p and p.Character)) then
                                    dataEsp.Highlight.Enabled = false
                                end
                            end)
                        else
                            dataEsp.Highlight.Enabled = false
                            dataEsp.Billboard.Enabled = false
                        end
                    elseif not dataEsp.Highlight.Enabled then
                    end
                end
            else
                if dataEsp.Highlight then dataEsp.Highlight.Enabled = false end
                if dataEsp.Billboard then dataEsp.Billboard.Enabled = false end
            end
        end
    end

    if active then
        for _, data in pairs(espData) do
            if data.Holder and data.Holder.Parent then
                local p = data.Plr
                local char = p and safeCloneref(p.Character)
                
                if char and isAlive(char) then
                    local root = char:FindFirstChild("HumanoidRootPart")
                    if root then
                        local distance = (root.Position - camPos).Magnitude
                        if distance <= maxDistance then
                            local screenPos, vis = cam:WorldToViewportPoint(root.Position + Vector3.new(0, 3.5, 0))
                            
                            if vis and canSee(root, char) then
                                data.IsVisible = true
                                data.Holder.Visible = true
                                data.Holder.ZIndex = math.clamp(1000 - math.floor(distance), 1, 1000)
                                data.Holder.Position = data.Holder.Position:Lerp(UDim2.new(0, screenPos.X, 0, screenPos.Y), dt * 18)
                            else
                                data.IsVisible = false
                                data.Holder.Visible = false
                            end
                        else
                            data.IsVisible = false
                            data.Holder.Visible = false
                        end
                    else
                        data.IsVisible = false
                        data.Holder.Visible = false
                    end
                else
                    data.IsVisible = false
                    if data.Holder then data.Holder.Visible = false end
                end
            end
        end
    end
end))
